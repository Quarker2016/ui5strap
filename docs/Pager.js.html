sap.ui.define(['./library', './ControlBase'], function(library, ControlBase){

	var Pager = ControlBase.extend("ui5strap.Pager", {
		metadata : {

			// ---- control specific ----
			library : "ui5strap",
			
			properties : {
				aligned : {
					type : "boolean",
					defaultValue : false
				},
				disabledPrevious : {
					type : "boolean",
					defaultValue : false
				},
				disabledNext : {
					type : "boolean",
					defaultValue : false
				}
			},
			aggregations : { 
				previous : {
					type : "ui5strap.Link",
					multiple:false
				}, 
				next : {
					type : "ui5strap.Link",
					multiple:false
				}
				
			}

		}
	}), PagerProto = Pager.prototype;
	
	/**
	 * @Protected
	 * @Override
	 */
	PagerProto._getStyleClassDesign = function(){
		var styleClass = " pager";
		return styleClass;
	};
	
	return Pager;
});
